#shopping cart programs with 1d approch
foods=[]
prices=[]
total=0
while True:
    food=input("enter the food to buy(q to exit):")
    if food.lower()=="q":
        break
    else:
        price=float(input("enter the price:$"))
        prices.append(price)
        foods.append(food)

for all_food in foods:
    print(all_food,end=" ")
for allPrice in prices:
    total +=allPrice

print(f"the total is :{total}")
